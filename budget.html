import pandas as pd
import requests
import io

import qgrid
qgrid.nbinstall(True) 

qgrid.set_defaults(remote_js=True, precision=4)

import matplotlib.pyplot as plt

url = "http://data.jerseycitynj.gov/dataset/fac622e5-f072-4900-ab77-966dd17855b3/resource/a00dda55-8c44-4590-a875-5d23057d37ab/download/budgetallcsv.csv"
s = requests.get(url).content
data = pd.read_csv(io.StringIO(s.decode('utf-8')))

qgrid.show_grid(data)

As you can see the data is poorly formatted. For one thing there are columns per year, making it difficult to do analysis across year. I reshape the data below to conform to more sane layout.

The prices also have a bunch of garbage in them (extra spaces, quotes, dollar signs and commas), to allow pandas to interpret the data as numeric I clean this up.

import numpy as np

melted_data = pd.melt(data, value_vars=['Adopted CY2011', 'Adopted CY2012', 'Adopted CY2013', 'Adopted CY2014', 'Adopted CY2015'], id_vars=['TYPE', 'SUB-TYPE', 'BUDGET CATEGORY', 'BUDGET SUB-CATEGORY', 'LINE ITEM'])
melted_data['variable'] = melted_data['variable'].str[-4:]

melted_data['value'] = melted_data['value'].str.replace(' ','')
melted_data['value'] = melted_data['value'].str.replace('$','')
melted_data['value'] = melted_data['value'].str.replace(',','')
melted_data['value'] = melted_data['value'].str.replace('-','')
melted_data.loc[melted_data['value'] == '','value'] = np.nan
melted_data['value'] = melted_data['value'].astype('float')
qgrid.show_grid(melted_data)

Now that I have a sane dataset I calculate the budget per-year.

def calc_budget(year):
    costs = year[year['TYPE'] == 'Appropriation']['value'].sum()
    revenue = year[year['TYPE'] == 'Revenue']['value'].sum()
    delta = revenue - costs
    return pd.Series({'costs': costs, 'revenue': revenue, 'delta': delta})

qgrid.show_grid(melted_data.groupby('variable').apply(func=calc_budget))

Two things stand out: we don't have costs for 2011, 2012's budget perfectly balanced (seems unlikely),and costs increase every year (makes sense).

def is_zero(df):
  zero = df[(df['value'] == 0) | pd.isnull(df['value'])]
  return zero
 
missing_data_by_year = melted_data.groupby('variable').apply(func=is_zero)
qgrid.show_grid(missing_data_by_year.groupby('variable').count())

The resulss are pretty much inline with expectations. It appears we are missing the same data every year (except for 2011 where we are missing all the cost data).

How that I have an idea of the shape and quality of the data I take a look and see who has changed the most over the past few years. Below I calculate highest range costs from 2012 to present. I filter out some garbage in the data (duplication of employee group healthcare categories).

remove_eleven = melted_data[melted_data['variable'] != '2011']#get rid of 2011 because costs are missing
no_missing = remove_eleven[(remove_eleven['value'] != 0) & (~pd.isnull(remove_eleven['value']))]

costs = no_missing[no_missing['TYPE'] == 'Appropriation']

def biggest_change(df):
    if df.shape[0] <= 1:
        return 0
    return (df['value'].max() - df['value'].min())

result = costs[costs['BUDGET CATEGORY'] != 'EMPLOYEE GROUP HEALTH INSURANCE'].groupby(['BUDGET SUB-CATEGORY', 'LINE ITEM']).apply(func=biggest_change)
qgrid.show_grid(result.reset_index())

Apparently projects offset by revenue have been decreasing over time (not really sure what that is).
